
        void getFibonacii(int a,int b, int n)
        {   
            int sum;
            if(n>0)
            {
                sum=a+b;
                a=b;
                b=sum;
                getFibonacii(a,b,n-1);
            }
        }
        int main()
        {
            n = 10;     
            a=0;        //first term
            b=1;        //second term

            //call function with (n-2) terms
            getFibonacii(a,b,n-2);  
            return 0;
        }
